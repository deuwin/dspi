produce(${name}_produceCargo,
    [
        $cycle_consume
    ],
    [
        $cycle_produce
    ],
    0
)

switch(FEAT_INDUSTRIES, SELF, ${name}_calcProduction, [
    setStockpileAverage($stockpile_level),
    $production_limit
    $produced_cargo
    setPowerDemand(GET_TEMP(PRODUCED))
]) { return ${name}_produceCargo; }

switch(FEAT_INDUSTRIES, SELF, ${name}_productionChange, [
    evalProduction(
        GET_PERM(STOCKPILE_AVERAGE),
        last_month_production("$output"),
        transported_last_month_pct("$output")
    )
]) { return; }

item(FEAT_INDUSTRIES, $name) {
    property {
        substitute: $id;
        override:   $id;
        cargo_types: [
            $accept_cargo
            $produce_cargo
        ];
    }
    graphics {
        build_prod_change:     initPermanentStorage(IND_SECONDARY, $id);
        extra_text_industry:   genExtraText();

        produce_cargo_arrival: noProduction;
        produce_256_ticks:     ${name}_calcProduction();

        stop_accept_cargo:     isStockpileFull;

        random_prod_change:    CB_RESULT_IND_PROD_NO_CHANGE;
        monthly_prod_change:   ${name}_productionChange();
    }
}
